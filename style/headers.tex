% 取消 book 类对 \chapter 的默认 \thispagestyle{plain} 行为
\makeatletter
\renewcommand\chapter{%
	\if@openright\cleardoublepage\else\clearpage\fi
	\thispagestyle{fancy}% 用 fancy 替代 plain
	\global\@topnum\z@
	\@afterindentfalse
	\secdef\@chapter\@schapter
}
\makeatother

% 避免 fancyhdr headheight 警告
\setlength{\headheight}{13pt}
\addtolength{\topmargin}{-1pt}

% 页眉格式
% 注意：具体内容（学位名称、论文标题）在正文前的摘要部分按需设置
% 此处仅做初始化，各内容文件可按需覆盖
\pagestyle{fancy}
\fancyhead{}
\fancyhead[CE]{\songti\zihao{5} \youreduction{}士学位论文}
\fancyhead[CO]{\songti\zihao{5} \yourtitlenamechinese}
\renewcommand{\headrulewidth}{1.5pt}
\renewcommand{\footrulewidth}{0pt}
